Conversion rules

How many times have you found yourself with two types that were almost equal?

Let's say you define Person: 

package main

import "fmt"

type Presenter interface {
	Introduce()
}


type Language struct {
    Name        string
    Maintainer  string
    Motivations []string
}


func NewLanguage(name, maintainer string, motivations []string) *Language {
	return &Language{
		Name:        name,
		Maintainer:  maintainer,
		Motivations: motivations,
	}
}
func (l *Language) Introduce() {
	fmt.Printf("%s is maintained by %s:", l.Name, l.Maintainer)
	fmt.Println("designed to provide:")
	for i, motivation := range l.Motivations {
		fmt.Printf("%d - %s\n", i+1, motivation)
	}
}

func main() {
	var presenter Presenter = NewLanguage(
		"Golang",
		"Google",
		[]string{
			"Native concurrency constructs",
			"Garbage collection",
			"Ease of adoption",
		},
	)

	presenter.Introduce()
}
